  Produkt = rep(produkty, each = length(sklepy) * length(miesiace)),
  Sklep = rep(sklepy, times = length(produkty) * length(miesiace)),

add_week_columns <- function(df, num_weeks) {
  col_name <- paste0(num_weeks, "00", sep='')
  df <- df %>%
    mutate(!!col_name := ymd(date.entered) + weeks(num_weeks - 1)) %>%
    relocate(!!col_name)
  return(df)
}
